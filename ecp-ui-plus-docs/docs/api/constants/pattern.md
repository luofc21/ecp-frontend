# Pattern 正则表达式

参考：

- [正则表达式30分钟入门教程](https://deerchao.cn/tutorials/regex/regex.htm)

## 基本用法

### 组件中使用

#### Options Api

- 从 `this.$constant.PATTERN` 获取

```vue
<template>
    <div>{{ $constant.PATTERN.xxx }}</div>
</template>
<script>
export default {
    mounted () {
        console.log(this.$constant.PATTERN);
    }
};
</script>
```

#### Composition Api

- 直接引入

```vue
<template>
    <div>{{ PATTERN.xxx }}</div>
</template>
<script setup>
import { PATTERN } from '@ecp/ecp-ui-plus';
console.log(PATTERN);
</script>
```

### JS 中使用

- 直接引入

```js
import { PATTERN } from '@ecp/ecp-ui-plus';
console.log(PATTERN);
```

## 常量枚举

```ts
export const PATTERN = {
    // 数字
    NUMBER: /[0-9]/,
    // 正整数
    POSITIVE_INT: /^[0-9]*[1-9][0-9]*$/,
    // 负整数
    NEGATIVE_INT: /^-[0-9]*[1-9][0-9]*$/,
    // 货币数字（小数点后两位）
    CURRENCY: /^\d+\.\d{1,2}$/,
    // 只允许中文匹配
    CHINESE: /[\u4e00-\u9fa5]/g,
    // 特殊字符
    SPECIALS: /^[^`~!@#$%^&*()+=|\\\][\]\{\}:;'\,.<>?]{1}[^`~!@$%^&+=\\\][\{\}:;'\,<>?]{0,}$/,

    // IP地址
    IP: /^([1-9]|[1-9]\d|1\d{2}|2[0-1]\d|22[0-3])(\.(\d|[1-9]\d|1\d{2}|2[0-4]\d|25[0-5])){3}$/,
    // Http IP地址
    HTTP_IP: /^http:\/\/([1-9]|[1-9]\d|1\d{2}|2[0-1]\d|22[0-3])(\.(\d|[1-9]\d|1\d{2}|2[0-4]\d|25[0-5])){3}$/,
    // 端口
    NETPORT: /^([0-9]|[1-9]\d|[1-9]\d{2}|[1-9]\d{3}|[1-5]\d{4}|6[0-4]\d{3}|65[0-4]\d{2}|655[0-2]\d|6553[0-5])$/,
    // 子网掩码
    MASK: /^(254|252|248|240|224|192|128|0)\.0\.0\.0|255\.(254|252|248|240|224|192|128|0)\.0\.0|255\.255\.(254|252|248|240|224|192|128|0)\.0|255\.255\.255\.(254|252|248|240|224|192|128|0)$/,
    // 物理地址
    MAC: /^[0-9,a-f,A-F]{2}[-][0-9,a-f,A-F]{2}[-][0-9,a-f,A-F]{2}[-][0-9,a-f,A-F]{2}[-][0-9,a-f,A-F]{2}[-][0-9,a-f,A-F]{2}$/,

    // 座机 + 移动电话
    PHONE: /^((0\d{2,3}-12345)|(0\d{2,3}-\d{7,8})|(0\d{2,3}-\d{7,8}-\d{3,4})|(0|86|17951)?(13[0-9]|15[012356789]|16[56]|17[012345678]|18[0-9]|19[0-9]|14[456789])[0-9]{8})$/,
    // 座机
    FIX_LINE: /^((0\d{2,3}-12345)|(0\d{2,3}-\d{7,8})|(0\d{2,3}-\d{7,8}-\d{3,4}))$/,
    // 移动电话
    MOBILEPHONE: /^((0|86|17951)?(13[0-9]|15[012356789]|16[56]|17[012345678]|18[0-9]|19[0-9]|14[456789])[0-9]{8})$/,
    // 传真
    FAX: /^(?:\d{3,4}-)?\d{7,8}(?:-\d{1,6})?$/,

    // email地址
    EMAIL: /^[\w-]+(\.[\w-]+)*@[\w-]+(\.[\w-]+)+$/,

    // 日期格式 YYYY-MM-DD
    DATE: /\d{4}-\d{2}-\d{2}$/,

    // 经度
    LONGITUDE: /^-?((180(\.0{0,6})?)|((\d{1,2}|1[0-7]\d)(\.\d{0,6})?))$/,
    // 纬度
    LATITUDE: /^-?((90(\.0{0,6})?)|((\d|[1-8]\d)(\.\d{0,6})?))$/,

    // 身份证字符格式
    ID_CARD: /(^\d{15}$)|(^\d{18}$)|(^\d{17}(\d|X|x)$)/,
    // 护照
    PASSPORT: /^(([PDGES]\d{8})|([PSD]E\d{7}))$/,
    // 港澳通行证
    HK_PASSPORT: /^[a-zA-Z0-9]{5,21}$/,
    // 行驶证-档案编号
    DRIVING_LICENSE: /^[0-9]{12}$/,
    // 行驶证-车辆识别代码（车架号）
    DRIVING_LICENSE_VIN: /^[A-HJ-NPR-Z0-9]{17}$/,
    // 行驶证-发动机号码
    DRIVING_LICENSE_ENGINE: /^[A-Z0-9]{12}$/,
    // 企业信用代码
    ENTERPRISE_CREDIT: /^([0-9A-HJ-NPQRTUWXY]{2}\d{6}[0-9A-HJ-NPQRTUWXY]{10}|[1-9]\d{14})$/,

    // 邮政编码
    POST_CODE: /[1-9]\d{5}(?!\d)/,

    // 车牌号码
    CAR_CODE: /^([京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领]{1}[A-Z]{1}(([0-9]{5}[DF])|([A-Z][A-HJ-NP-Z0-9])[0-9]{4}))$|^([京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领]{1}[A-Z]{1}[A-HJ-NP-Z0-9]{4}[A-HJ-NP-Z0-9挂学警港澳使领]{1})$/
};
```
